def show_src(src, idx=0, wavelet=None, endlim=None):
    wavelet = wavelet or src.data[:, idx]
    plt.figure()
    plt.plot(src.time_values, wavelet)
    plt.xlabel('Time (ms)')
    plt.ylabel('Amplitude')
    plt.tick_params()
    if endlim is not None:
        plt.xlim(0, endlim)
    plt.show()

from matplotlib import pyplot as plt
from matplotlib import cm
from mpl_toolkits.axes_grid1 import make_axes_locatable
def plot_shotrecord_vsp(rec, model, t0, tn, colorbar=True, true_limits=None):
    scale = np.max(rec) / 5.
    extent = [0, 0, 0, 0]
    if true_limits is None:
        extent = [t0, tn, model.origin[1] + model.domain_size[1], model.origin[1]]
    else:
        extent = true_limits
        
    plot =plt.imshow(rec, vmin=-scale, vmax=scale, cmap=cm.gray, extent=extent, aspect='equal')
    plt.xlabel('Time (ms)')
    plt.ylabel('Depth (m)')
    # Create aligned colorbar on the right
    if colorbar:
        ax = plt.gca()
        divider = make_axes_locatable(ax)
        cax = divider.append_axes("right", size="5%", pad=0.05)
        plt.colorbar(plot, cax=cax)
    plt.show()
